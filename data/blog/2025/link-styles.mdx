import { ExLinkStyle1, ExLinkStyle2 } from '@/components/blog.page/ExLinkStyle'

링크 스타일 여러개를 공유한 곳이 있는데, 거기서 맘에 드는 몇개를 적어두려 한다.  
나중에라도 더 찾게되면 추가해야 겠다.

<ExLinkStyle1 />

```html
<span
  className="relative before:absolute before:bottom-0 before:left-0 before:h-full before:w-full before:origin-bottom before:scale-y-[0.35] before:bg-sky-600 before:transition-transform before:duration-500 before:ease-in-out hover:before:scale-y-100"
>
  <span className="relative">Link Style 1</span>
</span>
```

<ExLinkStyle2 />

```html
<span
  className="group relative inline-block overflow-hidden rounded-lg bg-gray-200 px-6 py-3 [transform:translateZ(0)] before:absolute before:bottom-0 before:left-0 before:h-full before:w-full before:-translate-x-full before:bg-sky-600 before:transition before:duration-500 before:ease-in-out hover:before:translate-x-0"
>
  <span
    className="relative z-0 text-black transition duration-500 ease-in-out group-hover:text-gray-200"
  >
    Link Style 2
  </span>
</span>
```
